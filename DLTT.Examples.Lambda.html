{-# OPTIONS --safe #-}

module DLTT.Examples.Lambda where

open import DLTT.Base
open import DLTT.Examples.Sum

module _ {ℓ : Level} {𝕧i : ityp ℓ} {_≡?_ : 𝕧i → 𝕧i → 𝟚 {ℓ}} where
  𝕧 = Ground 𝕧i

  switchᵒ : ((𝕧 ⊕ 𝟙ᵒ) ⊕ 𝕧) ． → ltyp ℓ
  switchᵒ (inl (inl x)) = 𝟘ᵒ
  switchᵒ (inl (inr tt*)) = 𝟚ᵒ
  switchᵒ (inr x) = 𝟙ᵒ

  switchᵒsnd : (x : (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) ．) → List (switchᵒ x ．)
  switchᵒsnd (inl (inl _)) = 𝟘?
  switchᵒsnd (inl (inr tt*)) = 𝟚?
  switchᵒsnd (inr x) = 𝟙?

  lamfin : depfinltyp (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧)
  lamfin =  (switchᵒ , switchᵒsnd)

  Lambda : ltyp ℓ
  Lambda = Wᵒ (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) lamfin

  pattern var x f = sup (inl (inl x)) f
  pattern app f  = sup (inl (inr tt*)) f
  pattern lam x f  = sup (inr x) f

  freeocc : Lambda ． → 𝕧i → ℕ
  freeocc (var y f) x = ∣ y ≡? x ∣
  freeocc (app f) r = freeocc (f true) r + freeocc (f false) r
  freeocc (lam y f) x with y ≡? x
  ... | false = freeocc (f tt*) x
  ... | true = 0

  except : 𝕧i → (s : Lambda ．) → Fraction (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) lamfin s
  except x (var y f) = ∣ ¬ (y ≡? x) ∣ , λ ()
  except x (app f) = 1 , λ r → except x (f r)
  except x (lam y f) with y ≡? x
  ...| false = 1 , λ r → except x (f r)
  ...| true = 1 , λ r → const (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) lamfin (f r) 1


  lamsubstᵒ : (s : Lambda ．) (x : 𝕧i) (t : Lambda ．)
    → (s ∶ (Wº (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) & lamfin / (except x))) ⨾ (t ∶ Lambda) ^ freeocc s x ⊩ Lambda
  lamsubstᵒ (var y f) x t with y ≡? x
  ...| false = (var y f) , unitr ∘ unitr
  ...| true = t , unitl ∘ unitl ⨾ᶠ unitr
  lamsubstᵒ (app f) x t =
    (app λ r → lamsubstᵒ (f r) x t .fst) ,
    (unitr {Δ = ◇ ⨾ ◇ ⨾ ◇} ⨾ᶠ (lamsubstᵒ (f false) x t .snd ⨾ᶠ (unitr' ∘ lamsubstᵒ (f true) x t .snd)))
    ∘ ((id ⨾ᶠ assoc)
        ∘ (id ⨾ᶠ (id ⨾ᶠ (assoc' ∘ (swap ⨾ᶠ id) ∘ assoc)))
        ∘ (id ⨾ᶠ assoc')
        ∘ assoc'
        ∘ (id ⨾ᶠ (swap ∘ pow+ (freeocc (f true) x) (freeocc (f false) x))))
    ∘ ((id ⨾ᶠ id ⨾ᶠ unitr) ⨾ᶠ id)
  lamsubstᵒ (lam y f) x t with y ≡? x
  ... | false = (lam y (λ r → lamsubstᵒ (f r) x t .fst)) ,
    (id ⨾ᶠ swap) ∘ assoc' ∘ (id ⨾ᶠ (lamsubstᵒ (f tt*) x t .snd))
    ∘ (id ⨾ᶠ (id ⨾ᶠ unitl)) ∘ (id ⨾ᶠ assoc') ∘ assoc'
  ... | true = (lam y f) , const1 (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) lamfin (lam y f) ∘ (id ⨾ᶠ unitr' ∘ ((id ⨾ᶠ unitr) ∘ unitr))
