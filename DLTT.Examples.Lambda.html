module Prim where
  _≡?_ : LamVar {ℓ-zero} → LamVar {ℓ-zero} → 𝟚 {ℓ-zero}
  zero ≡? zero = true
  zero ≡? suc y = false
  suc x ≡? zero = false
  suc x ≡? suc y = x ≡? y

  _except_ : Lam {ℓ-zero} ． → LamVar {ℓ-zero} → sply ℓ-zero
  _except_ (var' y) x with y ≡? x
  ... | false = ι y
  ... | true = ◇
  _except_ (app' s t) x = (s except x) ⨾ (t except x)
  _except_ (lam' y s) x with y ≡? x
  ... | false = s except x
  ... | true = (lam' y s) ∶ Lam

  not : LamVar {ℓ-zero} → Lam {ℓ-zero} ． → 𝟚 {ℓ-zero}
  not x (var' y) = ¬ (y ≡? x)
  not x (app' s t) = not x s ∧ not x t
  not x (lam' y s) with y ≡? x
  ... | false = not x s
  ... | true = true

  uneq : LamVar {ℓ-zero} → Lam {ℓ-zero} ． → ℕ
  uneq x (var' y) = ∣ ¬ (y ≡? x) ∣
  uneq x (app' s t) = uneq x s + uneq x t
  uneq x (lam' y s) with y ≡? x
  ... | false = uneq x s
  ... | true = 1


  freeocc : Lam {ℓ-zero} ． → LamVar {ℓ-zero} → ℕ
  freeocc (var' y) x = ∣ y ≡? x ∣
  freeocc (app' s t) x = freeocc s x + freeocc t x
  freeocc (lam' y s) x with y ≡? x
  ... | false = freeocc s x
  ... | true = 0


  lamsubst : (s : Lam {ℓ-zero} ．) (x : LamVar {ℓ-zero}) (t : Lam {ℓ-zero} ．)
    → Lam {ℓ-zero} ．
  lamsubst (var' y) x t with y ≡? x
  ... | false = var' y
  ... | true = t
  lamsubst (app' s u) x t = app' (lamsubst s x t) (lamsubst u x t)
  lamsubst (lam' y s) x t with y ≡? x
  ... | false = lam' y (lamsubst s x t)
  ... | true = lam' y s


  lamsubstᵒ : (s : Lam {ℓ-zero} ．) (x : LamVar {ℓ-zero}) (t : Lam {ℓ-zero} ．)
    → (s except x) ⨾ (t ∶ Lam) ^ freeocc s x ⇒ lamsubst s x t ∶ Lam
    -- → (s ∶ Lam ∣ not x) ⨾ (t ∶ Lam) ^ freeocc s x ⇒ lamsubst s x t ∶ Lam
  lamsubstᵒ (var' y) x t with y ≡? x
  ... | false = {!!} -- unitr _
  ... | true = unitr ∘ unitl
  lamsubstᵒ (app' s u) x t = lamsubstᵒ s x t ⨾ᶠ lamsubstᵒ u x t ∘ {!!}
  lamsubstᵒ (lam' y s) x t with y ≡? x
  ... | false = lamsubstᵒ s x t
  ... | true = {!!} -- unitr _


module _ {ℓ : Level} {𝕧i : ityp ℓ} {_≡?_ : 𝕧i → 𝕧i → 𝟚 {ℓ}} where

  switchᵒ (inl (inl x)) = 𝟘ᵒ
  switchᵒ (inl (inr tt*)) = 𝟚ᵒ
  switchᵒ (inr x) = 𝟙ᵒ

  switchᵒsnd : (x : (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) ．) → List (switchᵒ x ．)
  switchᵒsnd (inl (inl _)) = 𝟘?
  switchᵒsnd (inl (inr tt*)) = 𝟚?
  switchᵒsnd (inr x) = 𝟙?

  -- switchᵒ .snd (inl (inr tt*)) = 𝟚?
  -- switchᵒ .snd (inr x) = 𝟙?
  lamfin : depfinltyp (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧)
  lamfin =  (switchᵒ , switchᵒsnd)

  Lambda : ltyp ℓ
  Lambda = Wᵒ (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) lamfin

  pattern var x f = sup (inl (inl x)) f
  pattern app f  = sup (inl (inr tt*)) f
  pattern lam x f  = sup (inr x) f

  freeocc : Lambda ． → 𝕧i → ℕ
  freeocc (var y f) x = ∣ y ≡? x ∣
  freeocc (app f) r = freeocc (f true) r + freeocc (f false) r
  freeocc (lam y f) x with y ≡? x
  ... | false = freeocc (f tt*) x
  ... | true = 0

  except x (app f) = 1 , λ r → except x (f r)

  ...| false = 1 , λ r → except x (f r)
  ...| true = 1 , λ r → const (𝕧 ⊕ 𝟙ᵒ ⊕ 𝕧) lamfin (f r) 1
