module _ {A : ltyp ℓ} {m : ℕ} {B : A ⇒' ltyp} {Δ₀ Δ₁ : sply ℓ}  where
  intro⊗ : ((a , _) : Δ₀ ⊩ A)
    → Δ₁ ⊩ B a
    → Δ₀ ^ m ⨾ Δ₁ ⊩ ⟨ x ∈ A ⟩^ m ⊗ B x
  intro⊗ (a , δ₀) (b , δ₁) = (a , b) , ⇒^ m δ₀ ⨾ᶠ δ₁

module _ {A : ltyp ℓ} {m : ℕ} {B : A ⇒' ltyp} {C : (Σᵒ A m B) ． → ltyp ℓ} {Δ₀ Δ₁ : sply ℓ}  where
  elim⊗ : ((p , _) : Δ₀ ⊩ ⟨ x ∈ A ⟩^ m ⊗ B x )
    → ((x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⊩ C (x , y))
    → Δ₁ ⨾ Δ₀ ⊩ C p
  elim⊗ ((a , b) , δ) f = f a b .fst , f a b .snd ∘ id ⨾ᶠ δ

  comp⊗ : ((p , _) : Δ₀ ⊩ ⟨ x ∈ A ⟩^ m ⊗ B x )
    → (c : (x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⊩ C (x , y))
    → (Δ₁ ⨾ Δ₀ ⇒ (elimΣ λ x y → c x y .fst) p ∶ C p)
  comp⊗ ((a , b) , δ) f = f a b .snd ∘ id ⨾ᶠ δ

  comp⊗' : ((p , _) : Δ₀ ⊩ ⟨ x ∈ A ⟩^ m ⊗ B x )
    → (c : (x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⊩ C (x , y))
    → ((x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⇒ (elimΣ λ x' y' → c x y .fst) p ∶ C (x , y))
  comp⊗' ((a , b) , δ) f x y = f x y .snd ∘ id ⨾ᶠ id


  -- ↓ this looks right
  comp : (c : (x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⊩ C (x , y))
    -- → ((x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⇒ (elimΣ {B = λ a → B a ．} λ x' y' → c x' y' .fst) (x , y) ∶ C (x , y))
    → ((x : A ．) (y : B x ．) → Δ₁ ⨾ (x ∶ A) ^ m ⨾ y ∶ B x ⇒ (c x y .fst) ∶ C (x , y))
  comp f x y = f x y .snd ∘ id ⨾ᶠ id


module _ {A : ltyp ℓ} {B : A ⇒' ltyp}  where
  sndᵒ : (z : (⟨ x ∈ A ⟩^ 0 ⊗ B x) ．) → (z ∶ ⟨ x ∈ A ⟩^ 0 ⊗ B x) ⊩ B (fst z)
  sndᵒ z = _by_ {A = B (fst z)}
    (elim⊗ {A = A} {0} {B} {C = λ z → B (fst z)} {z ∶ (⟨ x ∈ A ⟩^ 0 ⊗ B x)} {◇} -- {(x , y)} {λ _ y → y}
    (z , id) (λ x y → y , unitl ∘ unitl))
    unitl'
