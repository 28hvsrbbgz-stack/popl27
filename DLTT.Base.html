  [_,_] : sply ℓ → sply ℓ → sply ℓ

  data _⇒_ : sply ℓ → sply ℓ → ityp (ℓ-suc ℓ) where
    id : ∀ {Δ} → Δ ⇒ Δ
    _∘_ : ∀ {Δ₀ Δ₁ Δ₂} → Δ₁ ⇒ Δ₂ → Δ₀ ⇒ Δ₁ → Δ₀ ⇒ Δ₂
    _⨾ᶠ_ : ∀ {Δ₀ Δ₁ Δ₂ Δ₃} → Δ₀ ⇒ Δ₁ → Δ₂ ⇒ Δ₃ → Δ₀ ⨾ Δ₂ ⇒ Δ₁ ⨾ Δ₃

    swap : ∀ {Δ₀ Δ₁} → Δ₀ ⨾ Δ₁ ⇒ Δ₁ ⨾ Δ₀
    unitr : ∀ {Δ} → Δ ⨾ ◇ ⇒ Δ
    unitr' : ∀ {Δ} → Δ ⇒ Δ ⨾ ◇
    assoc : ∀ {Δ₀ Δ₁ Δ₂} → Δ₀ ⨾ (Δ₁ ⨾ Δ₂) ⇒ (Δ₀ ⨾ Δ₁) ⨾ Δ₂

    curry : ∀ {Δ₀ Δ₁ Δ₂} →  (Δ₀ ⨾ Δ₁) ⇒ Δ₂ → Δ₀ ⇒ [ Δ₁ , Δ₂ ]
    uncurry : ∀ {Δ₀ Δ₁ Δ₂} → Δ₀ ⇒ [ Δ₁ , Δ₂ ] →  (Δ₀ ⨾ Δ₁) ⇒  Δ₂

    bind : {Δ : sply ℓ} {(A , Θ) : Σ⟨ A ∶ ityp ℓ ⟩ (A → sply ℓ)}
      → ((x : A) → Δ ⇒ Θ x) → (Δ ⇒ Λ A Θ)
    free : {Δ : sply ℓ}  {(A , Θ) : Σ⟨ A ∶ ityp ℓ ⟩ (A → sply ℓ)}
      → (Δ ⇒ Λ A Θ) → ((x : A) → Δ ⇒ Θ x)

    !ᶠ : ∀ {Δ₀ Δ₁} → Δ₀ ⇒ Δ₁ → ! Δ₀ ⇒ ! Δ₁
    use : ∀ {Δ} → ! Δ ⇒ Δ
    mult : ∀ {Δ} → ! Δ ⇒ ! (! Δ)
    coh◇ : ◇ ⇒ ! ◇
    coh⨾ : ∀ {Δ₀ Δ₁} → ! Δ₀ ⨾ ! Δ₁ ⇒ ! (Δ₀ ⨾ Δ₁)
    coh⨾' : ∀ {Δ₀ Δ₁} → ! (Δ₀ ⨾ Δ₁) ⇒ ! Δ₀ ⨾ ! Δ₁

    dupl : ∀ {Δ} → ! Δ ⇒ ! (Δ ⨾ Δ)
    erase : ∀ {Δ} → ! Δ ⇒ ◇

infixl 25 Λ
syntax Λ A (λ x → Δ) = Λ⟨ x ∶ A ⟩ Δ



unitl : ∀ {Δ : sply ℓ} → ◇ ⨾ Δ ⇒ Δ
unitl = unitr ∘ swap

unitl' : ∀ {Δ : sply ℓ} → Δ ⇒ ◇ ⨾ Δ
unitl' = swap ∘ unitr'



assoc' : ∀ {Δ₀ Δ₁ Δ₂ : sply ℓ} → (Δ₀ ⨾ Δ₁) ⨾ Δ₂ ⇒ Δ₀ ⨾ (Δ₁ ⨾ Δ₂)
assoc' = swap
  ∘ assoc
  ∘ id ⨾ᶠ swap
  ∘ swap
  ∘ swap ⨾ᶠ id
  ∘ assoc
  ∘ swap


eval : ∀{Δ₀ Δ₁ : sply ℓ} → [ Δ₀ , Δ₁ ] ⨾ Δ₀ ⇒ Δ₁
eval {_} {Δ₀} {Δ₁} = uncurry (id {Δ = [ Δ₀ , Δ₁ ]})


_^_ : sply ℓ → ℕ → sply ℓ
Δ ^ zero = ◇
Δ ^ (suc m) = Δ ⨾ (Δ ^ m)


infixr 50 _^_

⇒^ : ∀{ℓ}{Δ₀ Δ₁ : sply ℓ} → (m : ℕ) → Δ₀ ⇒ Δ₁ → (Δ₀ ^ m) ⇒ (Δ₁ ^ m)
⇒^ zero δ = id
⇒^ (suc m) δ = δ ⨾ᶠ ⇒^ m δ

^asrec : sply ℓ → ℕ → sply ℓ
^asrec {ℓ} Δ = natElim {A = λ _ → sply ℓ} (λ _ → _⨾ Δ) ◇

⇒^asrec : ∀{ℓ}{Δ₀ Δ₁ : sply ℓ} → Δ₀ ⇒ Δ₁ → (m : ℕ) → (^asrec Δ₀ m) ⇒ (^asrec Δ₁ m)
⇒^asrec {ℓ} {Δ₀} {Δ₁} δ = natElim {A = λ m → (^asrec Δ₀ m) ⇒ (^asrec Δ₁ m)} (λ _ → _⨾ᶠ δ) (id)

∣_∣ : 𝟚 {ℓ} → ℕ
∣ false ∣ = 0
∣ true ∣ = 1

⨂ : (A : ltyp ℓ) → (List (A ．)) → (Δ : A ． → sply ℓ) → sply ℓ
⨂ A [] Δ = ◇
⨂ A (a ∷ as) Δ = Δ a ⨾ ⨂ A as Δ

⨂ᶠ : {A : ltyp ℓ} {as : List (A ．)} {Δ₀ Δ₁ : A ． → sply ℓ}
  → ((x : A ．) → Δ₀ x ⇒ Δ₁ x)
  → ⨂ A as Δ₀ ⇒ ⨂ A as Δ₁
⨂ᶠ {as = []} f = id
⨂ᶠ {as = a ∷ as} f = f a ⨾ᶠ ⨂ᶠ {as = as} f

⨂lax : {A : ltyp ℓ} {as : List (A ．)} {Δ₀ Δ₁ : A ． → sply ℓ}
  → ⨂ A as Δ₀ ⨾ ⨂ A as Δ₁ ⇒ ⨂ A as (λ x → Δ₀ x ⨾ Δ₁ x)
⨂lax {as = []} = unitl
⨂lax {as = a ∷ as} = id ⨾ᶠ ⨂lax {as = as} ∘ assoc ∘ id ⨾ᶠ assoc' ∘ id ⨾ᶠ (swap ⨾ᶠ id) ∘ id ⨾ᶠ assoc ∘ assoc'



data LamVar {ℓ : Level}  : Set ℓ where
  zero : LamVar
  suc  : (n : LamVar {ℓ}) → LamVar



data LamTerm {ℓ : Level} : ityp ℓ where
  var' : LamVar {ℓ} → LamTerm
  app' : LamTerm {ℓ} → LamTerm {ℓ} → LamTerm
  lam' : LamVar {ℓ} → LamTerm {ℓ} → LamTerm


Wi :  (A : ltyp ℓ) (B : A ⇒' finltyp) → ityp ℓ
Wi A B = W (A ．) (λ x → B .fst x ．)

module _ (A : ltyp ℓ) (B : A ⇒' finltyp) where
  Fraction : Wi A B → ityp ℓ
  Fraction = recW (λ a g → ℕ × ((y : B .fst a ．) → g y))
  -- (sup a f) = ℕ × ((y : B .fst a ．) → Fraction (f y))

  fractsply : (x : Wi A B) → Fraction x → sply ℓ
  fractsply (sup a f) (m , mf) = (a ∶ A) ^ m ⨾ ⨂ (B .fst a) (B .snd a) λ y → fractsply (f y) (mf y)

  const : (x : Wi A B) → ℕ → Fraction x
  const (sup a f) m = m , (λ y → const (f y) m)




module _ (A : ltyp ℓ) where
  ListFraction : List (A ．) → ityp ℓ
  ListFraction = foldr (λ _ → ℕ ×_) 𝟙
  -- ListFraction [] = 𝟙
  -- ListFraction (x ∷ xs) = ℕ × ListFraction xs

  Listsply : (xs : List (A ．)) → ListFraction xs → sply ℓ
  Listsply [] m = ◇
  Listsply (x ∷ xs) (m , mf) = (x ∶ A) ^ m ⨾ Listsply xs mf

  𝟘ᵒ : ltyp ℓ

  Wᵒ : (A : ltyp ℓ) → (A ⇒' finltyp) → ltyp ℓ
  Wº_&_/_ : (A : ltyp ℓ) → (B : A ⇒' finltyp) → ((a : Wi A B) → Fraction A B a) → ltyp ℓ
  Listᵒ : (A : ltyp ℓ) → ltyp ℓ
  Listᵒ_/_ : (A : ltyp ℓ) → ((xs : List (A ．)) → ListFraction A xs) → ltyp ℓ
  Lam : ltyp ℓ


⟦ Ground A ⟧ = A , λ x → ι x
⟦ 𝟘ᵒ ⟧ = 𝟘 , λ ()
⟦ 𝟙ᵒ ⟧ = 𝟙 , (λ _ → ◇)
⟦ 𝟚ᵒ ⟧ = 𝟚 , (λ _ → ◇)
⟦ Πᵒ A m B ⟧ = ((x : A ．) → B x ．) , λ f → Λ⟨ x ∶ A ． ⟩ [ (x ∶ A) ^ m , f x ∶ B x ]
⟦ Π! A B ⟧ = ((x : A ．) → B x ．) , λ f → Λ⟨ x ∶ A ． ⟩ [ ! (x ∶ A) , f x ∶ B x ]
⟦ Σᵒ A m B ⟧ = Σ (A ．) (λ x → B x ．) , λ (x , y) → (x ∶ A) ^ m ⨾ y ∶ B x
⟦ Wᵒ A (B , Bfin) ⟧ = W (A ．) (λ x → B x ．) , recW (λ x Δ → (x ∶ A) ⨾ ⨂ (B x) (Bfin x) Δ )
⟦_⟧ {ℓ} (Wº_&_/_  A B m) = W (A ．) (λ x → B .fst x ．) , λ x → fractsply A B x (m x)


⟦ Listᵒ A ⟧ = List (A ．) , foldr (λ x → x ∶ A ⨾_) ◇
⟦ Listᵒ A / m ⟧ = List (A ．) , λ xs → Listsply A xs (m xs)

  LamSply : LamTerm {ℓ} → sply ℓ
  LamSply (var' x) = ι x
  LamSply (app' s t) = LamSply s ⨾ LamSply t
  LamSply (lam' x s) = LamSply s -- ι x ⨾ LamSply s
