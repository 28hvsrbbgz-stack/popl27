Finᵒ : ℕ → ltyp ℓ
Finᵒ zero = 𝟘ᵒ
Finᵒ (suc n) = 𝟙ᵒ ⊕ Finᵒ n

Finfree : (n : ℕ) → (k : Finᵒ {ℓ = ℓ} n ．) → ◇ ⇒ k ∶ Finᵒ n
Finfree (suc n) (inl snd₁) = unitl'
Finfree (suc n) (inr k) = unitl' ∘ Finfree n k


module _
  {A : ltyp ℓ} {B : A ⇒' finltyp}
  where
  compspWᵒ : (p : (Wᵒ A B) ．)
    → (p ∶ Wᵒ A B) ≡ elimW (λ x f Δ → x ∶ A ⨾ ⨂ (B .fst x) (B .snd x) Δ) p
  compspWᵒ f = refl

module _
  {A : ltyp ℓ} {(B , bs) : A ⇒' finltyp}
  {C : Wᵒ A (B , bs) ⇒' ltyp}
  {c : (a : A ．) → (f : B a ． → Wᵒ A (B , bs) ．) → ((y : B a ．) → C (f y) ．) → C (sup a f) ．}
  {Δ₁ : A ． → sply ℓ}
  where
  elimWᵒ' :
    ({x : A ．} {f : B x ． → Wᵒ A (B , bs) ．} {g : (y : B x ．) → C (f y) ．} →
      (x ∶ A)  ⨾ Δ₁ x ⨾ ⨂ (B x) (bs x) (λ y → g y ∶ C (f y)) ⇒ c x f g ∶ C (sup x f))
    → (w : Wᵒ A (B , bs) ．)
    → w ∶ Wᵒ A (B , bs) ⨾ recW (λ a Δ' → Δ₁ a ⨾ ⨂ (B a) (bs a) Δ') w ⇒ elimW c w ∶ C w
  elimWᵒ' δ = elimW (λ x f g → δ {x} ∘
    (id ⨾ᶠ (id ⨾ᶠ (⨂ᶠ g ∘ ⨂lax)
      ∘ assoc' ∘ swap ⨾ᶠ id ∘ assoc) ∘ assoc'))

  elimWᵒ :
    {Δ₀ : sply ℓ} {p : Wᵒ A (B , bs) ．} →
    ({x : A ．} {f : B x ． → Wᵒ A (B , bs) ．} {g : (y : B x ．) → C (f y) ．} →
      (x ∶ A) ⨾ Δ₁ x ⨾ ⨂ (B x) (bs x) (λ y → g y ∶ C (f y)) ⇒ c x f g ∶ C (sup x f))
      → (Δ₀ ⇒ p ∶ Wᵒ A (B , bs))
    → Δ₀ ⨾ recW (λ a Δ' → Δ₁ a ⨾ ⨂ (B a) (bs a) Δ') p ⇒ elimW c p ∶ C p
  elimWᵒ {p = p} δ δ₁ = elimWᵒ' δ p ∘ δ₁ ⨾ᶠ id

module _
  {A : ltyp ℓ} {(B , bs) : A ⇒' finltyp}
  {C : Wᵒ A (B , bs) ⇒' ltyp}
  {c : (a : A ．) → (f : B a ． → Wᵒ A (B , bs) ．) → ((y : B a ．) → C (f y) ．) → C (sup a f) ．}
  {Δ₁ : A ． → sply ℓ}
  where
  compWᵒ :
    ({x : A ．} {f : B x ． → Wᵒ A (B , bs) ．} {g : (y : B x ．) → C (f y) ．} →
      (x ∶ A) ⨾ Δ₁ x ⨾ ⨂ (B x) (bs x) (λ y → g y ∶ C (f y)) ⇒ c x f g ∶ C (sup x f))

      → {a : A ．}
      → {f : B a ． → Wᵒ A (B , bs) ．}
      → {Δ₀ : sply ℓ}
      → (Δ₀ ⇒ a ∶ A)
      → Δ₀  ⨾ Δ₁ a ⨾ ⨂ (B a) (bs a) (λ y → elimW {P = λ y → C y ．} c (f y) ∶ C (f y)) ⇒
        c a f (λ y → elimW {P = λ z → C z ．} c (f y))
        -- elimW {P = λ z → C z ．} c (sup a f)
          ∶ C (sup a f)
  compWᵒ δ {a} {f} δ₀ = δ {g = (λ y → elimW {P = λ y → C y ．} c (f y))} ∘ δ₀ ⨾ᶠ id
