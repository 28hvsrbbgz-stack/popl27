module DLTT.Structural where

open import DLTT.Base


private
  variable
    ℓ ℓ' : Level
    Δ₀ Δ₁ : sply ℓ

Idᵒ : (A : ltyp ℓ)
 → (a : A ．)
 → a ∶ A ⊩ A
Idᵒ A x = x , id

Cut : {A B : ltyp ℓ}
  {m : ℕ}
  → ((a , _) : Δ₁ ⊩ A)
  → Δ₀ ⨾ (a ∶ A) ^ m ⊩ B
  → Δ₀ ⨾ Δ₁ ^ m ⊩ B
Cut (a , δ₀) (b , δ₁) = b , δ₁ ∘ id ⨾ᶠ ⇒^ _ δ₀

Cut' : {A B : ltyp ℓ}
  {m : ℕ} {a : A ．} {b : B ．}
  → (Δ₁ ⇒ a ∶ A)
  → Δ₀ ⨾ (a ∶ A) ^ m ⇒ b ∶ B
  → Δ₀ ⨾ Δ₁ ^ m ⇒ b ∶ B
Cut' δ₀ δ₁ = δ₁ ∘ id ⨾ᶠ ⇒^ _ δ₀


Coerce : {A : ltyp ℓ}
  (δ : Δ₁ ⇒ Δ₀)
  → Δ₀ ⊩ A
  → Δ₁ ⊩ A
Coerce δ₀ (a , δ₁) = a , (δ₁ ∘ δ₀)


Convsply : {A : ltyp ℓ}
  → Δ₀ ⊩ A
  → Δ₀ ≡ Δ₁
  → Δ₁ ⊩ A
Convsply (a , δ) refl = a , δ
